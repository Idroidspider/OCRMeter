Usage

Titanium Appcelerator usage

var OCRparse = require('lib/OcrParse');
var ocr = new OCRparse();
//get picture and value
ocr.getValue(function(image, value, number) {
				imageView.image = image;//image blob
				valueLabel.text = number;//parsed number eg. 1234.5
				valueLabel._value = value;//parsed value string eg. 12345
			});
/*Update value only*/
ocr.updateValue(valueLabel._value, function(value, number) {
				valueLabel.text = number;//parsed number eg. 1234.5
				valueLabel._value = value;//parsed value string eg. 12345
			   });

Server API

Get server status
GET http://host:port/api/getStatus
Output params: 
{
  "version": "0.9.0",
  "debug": "on",
  "status": "running"
}
Get API version
GET http://host:port/api/getVersion
Output params: 
{
  "version": "0.9.0"
}
Get value from picture
POST http://host:port/api/parseValue
Input params:
  • image - image file
  • decimalPlaces - decimal places of value (default 1)
  • valueLength - digits count (default 7)
Output params: 
{
	status: "done",
	ocrValues: [
		"0064750",
		"0054759",
		"9054754",
		"9954754",
		"9954754",
		"9054754"
	],
	value: "9054754",
	number: 905475.4
}
